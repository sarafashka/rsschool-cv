RGB To Hex Conversion
function rgb(r, g, b){
                                        const rgbArr = [r, g, b].map(item => {
                                            if (Number(item) < 0) {
                                                item = 0;
                                            } else if (Number(item) > 255) {
                                                item = 255;
                                            } return item;
                                        });
                                        const hex = rgbArr.map(item => item.toString(16));
                                        const hexFull = hex.map(item => {
                                            if (item.length === 1) {
                                                item = `0${item}`;
                                            } return item;
                                        });
                                        return hexFull.join('').toUpperCase();
                                    };
